Atualizar senha :

UPDATE public.usuarios
SET senha_hash = crypt('SUA_NOVA_SENHA_AQUI', gen_salt('bf'))
WHERE id = 9; -- Substitua pelo ID do usuário que deseja alterar

Excluir usuário:

DELETE FROM public.usuarios 
WHERE id = 9; -- Substitua o 9 pelo ID do usuário que quer apagar
